PREFIX co: <http://www.semanticweb.org/kde_project/covid_effects#>

select DISTINCT * {

    {  

    select DISTINCT ?Attraction_Countyname (count(distinct ?att_name) as ?Attraction_Count) where { 

    ?s co:att_is_in_cvd_countyname ?Attraction_Countyname .

    ?s co:att_name ?att_name .

    }GROUP BY ?Attraction_Countyname order by desc(?Attraction_Count)  

    }

    

    {

    select DISTINCT ?Accomodation_Countyname (count(distinct ?acc_name) as ?Accomodation_Count)         where {    

    ?s co:acc_is_in_cvd_countyname ?Accomodation_Countyname .

    ?s co:acc_name ?acc_name .

    }GROUP BY ?Accomodation_Countyname order by desc(?Accomodation_Count) 

    }   

}
